```R
# 样本相关性热图
# 先计算样本和样本之间的相关系数，再画成热图

library(pheatmap)

data <- read.csv("G:/thermal_protemic/preprocessed_proteomics_24h.csv",
                 row.names = 1,
                 check.names = FALSE)

# 检查数据是否为数值型
data <- as.data.frame(data)
data[] <- lapply(data, as.numeric)

#计算样本相关性矩阵，这里用的是Pearson相关系数
cor_matrix <- cor(data, method = "pearson")
#查看相关性矩阵
round(cor_matrix, 3)


# 加样本分组注释,方便画图
annotation_col <- data.frame(
  Group = c("Ctrl", "Ctrl", "Ctrl", "Treat", "Treat", "Treat")
)

rownames(annotation_col) <- colnames(data)

# 画热图
#pheatmap(cor_matrix),下面是美化版

pheatmap(cor_matrix,
         display_numbers = TRUE,
         number_format = "%.2f",
         fontsize_number = 10,
         annotation_col = annotation_col,
         annotation_row = annotation_col,
         main = "Sample Correlation Heatmap - 24h")

png("G:/thermal_protemic/24h/Correlation_heatmap_24h.png",
    width = 1800, height = 1600, res = 300)
```
