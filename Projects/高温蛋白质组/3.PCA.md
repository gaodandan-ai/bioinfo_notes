library(ggplot2)

data <- read.csv("G:/thermal_protemic/preprocessed_proteomics_data_cutf.csv", #文件路径改这里
                 row.names = 1,
                 check.names = FALSE)


# PCA要求行是样本，列是变量，但目前行是蛋白，列是样本，所以要转置
data_t <- t(data)

#检查数据类型，避免报错
data_t <- as.data.frame(data_t)
data_t[] <- lapply(data_t, as.numeric)

# 做PCA
pca <- prcomp(data_t, scale. = TRUE)

# 画PCA图

pca_df <- as.data.frame(pca$x)

pca_df$group <- group
pca_df$sample <- rownames(pca_df)

ggplot(pca_df, aes(x = PC1, y = PC2, color = group, label = sample)) +
  geom_point(size = 4) +
  geom_text(vjust = -1) +
  theme_minimal() +
  labs(title = "PCA - all time Proteomics",#名称改这里
       x = paste0("PC1: ", round(summary(pca)$importance[2,1]*100, 2), "%"),
       y = paste0("PC2: ", round(summary(pca)$importance[2,2]*100, 2), "%"))




# 保存图片

ggsave("G:/thermal_protemic/PCA_all.png", width = 6, height = 5, dpi = 300)#保存路径改这里
