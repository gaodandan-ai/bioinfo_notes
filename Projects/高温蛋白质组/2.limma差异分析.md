library(limma)

data <- read.csv("G:/thermal_protemic/preprocessed_proteomics_1h.csv", 
                 row.names = 1,
                 check.names = FALSE)
#检查数据结构
head(data)
colnames(data)
dim(data)

group <- factor(c("Ctrl","Ctrl","Ctrl","Treat","Treat","Treat"))

#4.limma差异分析

design <- model.matrix(~0 + group)
colnames(design) <- c("Ctrl","Treat")

fit <- lmFit(data, design)

contrast.matrix <- makeContrasts(Treat - Ctrl, levels = design)

fit2 <- contrasts.fit(fit, contrast.matrix)
fit2 <- eBayes(fit2)

#5.提取结果
result <- topTable(fit2, number = Inf)

head(result)

#6.筛选差异蛋白

deg <- result[
  abs(result$logFC) > 1 & result$adj.P.Val < 0.05,
]
# 保存结果
write.csv(result, "G:/thermal_protemic/1h/1h_all_results.csv")
write.csv(deg, "G:/thermal_protemic/1h/1h_DEG.csv")

# ===================== 1. 读取文件 =====================
deg1h <- read.csv("G:/thermal_protemic/1h/1h_DEG.csv", row.names=1, stringsAsFactors=FALSE)
deg4h <- read.csv("G:/thermal_protemic/4h/4h_DEG.csv", row.names=1, stringsAsFactors=FALSE)
deg24h <- read.csv("G:/thermal_protemic/24h/24h_DEG.csv", row.names=1, stringsAsFactors=FALSE)

# ===================== 2. 提取蛋白ID列表 =====================
list1h <- rownames(deg1h)
list4h <- rownames(deg4h)
list24h <- rownames(deg24h)

# 打包成列表（给韦恩图用）
deg_list <- list(
  `1h` = list1h,
  `4h` = list4h,
  `24h` = list24h
)

# ===================== 3. 画韦恩图（最简单版本） =====================
# 如果没装包，先运行下面这行：
# install.packages("VennDiagram")

library(VennDiagram)

venn.diagram(
  x = deg_list,
  filename = "G:/thermal_protemic/DEG_venn.png",  # 保存路径
  fill = c("red", "blue", "green"),               # 三个颜色
  alpha = 0.5,                                    # 透明度
  cat.cex = 1.5,                                  # 标签大小
  main = "DEG overlap 1h vs 4h vs 24h",           # 标题
  main.cex = 2
)
