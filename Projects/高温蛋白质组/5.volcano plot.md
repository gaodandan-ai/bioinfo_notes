library(ggplot2)


result <- read.csv("G:/thermal_protemic/1h/1h_all_results.csv",
                 row.names = 1,
                 check.names = FALSE)

# 添加分组标签
result$group <- "Not Sig"

result$group[result$logFC > 1 & result$adj.P.Val < 0.05] <- "Up"
result$group[result$logFC < -1 & result$adj.P.Val < 0.05] <- "Down"

# 画火山图

ggplot(result, aes(x = logFC, y = -log10(adj.P.Val), color = group)) +
  geom_point(size = 2, alpha = 0.7) +
  scale_color_manual(values = c("blue", "grey", "red")) +
  geom_vline(xintercept = c(-1, 1), linetype = "dashed") +
  geom_hline(yintercept = -log10(0.05), linetype = "dashed") +
  theme_classic() +
  labs(title = "Volcano Plot - 1h Heat Stress",
       x = "log2 Fold Change",
       y = "-log10(adj.P.Val)")

ggsave("G:/thermal_protemic/1h/Volcano_1h.png",
       width = 6, height = 5, dpi = 300)
