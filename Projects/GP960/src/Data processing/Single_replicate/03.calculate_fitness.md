#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
NaCl NaCl胁迫耐受性分析
分析基因过表达实验中哪些基因在胁迫条件下表现出明显的耐受性。

分析指标：
- 生长性能指标 ：最大生长速率、最终生物量（OD最大值）、滞后期（OD达到初始值两倍的时间）、生长曲线下面积(AUC)
- 胁迫耐受性指标 ：相对适应性(Relative Fitness)、综合相对适应性评分
- 筛选标准 ：综合相对适应性 > 1.05, 最终OD相对适应性 > 1.05
- 非胁迫不耐受基因 ：综合相对适应性 < 0.95, 最终OD相对适应性 < 0.95

输出：
- 胁迫耐受性基因排名列表
- 非胁迫不耐受基因排名列表
- 生长指标统计表
- 可视化图表
"""

import pandas as pd
import numpy as np
import matplotlib.pyplot as plt
import seaborn as sns
from scipy import stats
from scipy.optimize import curve_fit
from pathlib import Path
import warnings
warnings.filterwarnings('ignore')

# 设置matplotlib参数，避免中文显示问题
plt.rcParams['font.family'] = ['DejaVu Sans', 'sans-serif']
plt.rcParams['axes.unicode_minus'] = False

class StressToleranceAnalyzer:
    def __init__(self, data_dir, rf_threshold_stress=1.05, comprehensive_rf_threshold_stress=1.05, 
                 rf_threshold_nonstress=0.95, comprehensive_rf_threshold_nonstress=0.95):
        """初始化分析器"""
        self.data_dir = Path(data_dir)
        self.results = {}
        self.rf_threshold_stress = rf_threshold_stress
        self.comprehensive_rf_threshold_stress = comprehensive_rf_threshold_stress
        self.rf_threshold_nonstress = rf_threshold_nonstress
        self.comprehensive_rf_threshold_nonstress = comprehensive_rf_threshold_nonstress    
        
    def load_data(self):
        """加载数据"""
        print("正在加载清理后的数据...")
        
        # 加载四个清理后的数据文件
        cleaned_data_dir = self.data_dir
        self.mutant_stress = pd.read_csv(cleaned_data_dir / f"01.mutant_stress_OD_cleaned.csv", index_col=0)
        self.mutant_nonstress = pd.read_csv(cleaned_data_dir / f"02.mutant_nonstress_OD_cleaned.csv", index_col=0)
        self.ctrl_stress = pd.read_csv(cleaned_data_dir / f"03.Ctrol_stress_OD_cleaned.csv", index_col=0)
        self.ctrl_nonstress = pd.read_csv(cleaned_data_dir / f"04.Ctrol_nonstress_OD_cleaned.csv", index_col=0)
        
        # 获取时间点
        self.time_points = self.mutant_stress.index.values
        
        # 获取基因列表（去除空格）
        self.genes = [col.strip() for col in self.mutant_stress.columns]
        
        print(f"数据加载完成:")
        print(f"- 时间点: {len(self.time_points)} 个 ({self.time_points[0]:.1f} - {self.time_points[-1]:.1f} 分钟)")
        print(f"- 基因数: {len(self.genes)} 个")
        
    def logistic_growth(self, t, K, r, t0):
        """Logistic生长模型"""
        return K / (1 + np.exp(-r * (t - t0)))
    
    def calculate_growth_parameters(self, od_values, time_points):
        """计算生长参数"""
        try:
            # 移除异常值
            od_values = np.array(od_values)
            valid_idx = (od_values > 0) & (od_values < 20) & np.isfinite(od_values)
            
            if np.sum(valid_idx) < 10:  # 数据点太少
                return None
                
            od_clean = od_values[valid_idx]
            time_clean = time_points[valid_idx]
            
            # 拟合Logistic生长曲线
            try:
                # 初始参数估计
                K_init = np.max(od_clean)
                r_init = 0.01
                t0_init = np.median(time_clean)
                
                popt, _ = curve_fit(self.logistic_growth, time_clean, od_clean, 
                                  p0=[K_init, r_init, t0_init], 
                                  maxfev=1000, bounds=([0.1, 0.001, 0], [20, 1, 3000]))
                
                K, r, t0 = popt
                
            except:
                # 如果拟合失败，使用简单方法
                K = np.max(od_clean)
                r = 0.01
                t0 = time_clean[np.argmax(np.gradient(od_clean))]
            
            # 计算各项指标
            final_od = od_clean[-1]
            max_growth_rate = r * K / 4  # Logistic模型的最大生长速率
            
            # 滞后期：OD达到初始值2倍的时间
            lag_time = 0
            initial_od = od_clean[0]
            threshold = initial_od * 2
            lag_idx = np.where(od_clean >= threshold)[0]
            if len(lag_idx) > 0:
                lag_time = time_clean[lag_idx[0]]
            
            # 生长曲线下面积 (AUC)
            auc = np.trapz(od_clean, time_clean)
            
            return {
                'final_od': final_od,
                'max_growth_rate': max_growth_rate,
                'lag_time': lag_time,
                'auc': auc,
                'carrying_capacity': K,
                'growth_rate_constant': r,
                'inflection_point': t0
            }
            
        except Exception as e:
            print(f"计算生长参数时出错: {e}")
            return None
    
    def find_control_column(self, gene):
        """查找基因对应的对照组列名"""
        # 在对照组文件中查找包含该基因的列
        for col in self.ctrl_stress.columns:
            if gene in col:
                return col
        return None
    
    def analyze_gene_performance(self):
        """分析每个基因的生长性能（包含对照组数据计算相对适应性）"""
        print("正在分析基因生长性能（包含对照组数据）...")
        
        gene_results = []
        
        for i, gene in enumerate(self.genes):
            if i % 500 == 0:
                print(f"处理进度: {i+1}/{len(self.genes)}")
            
            try:
                # 获取该基因在过表达组的数据
                gene_col = gene + '  ' if gene + '  ' in self.mutant_stress.columns else gene
                
                if gene_col not in self.mutant_stress.columns:
                    continue
                
                # 获取过表达组数据
                mut_stress_data = self.mutant_stress[gene_col].values
                mut_nonstress_data = self.mutant_nonstress[gene_col].values
                
                # 查找对照组数据
                ctrl_col = self.find_control_column(gene)
                if ctrl_col is None:
                    continue
                
                con_stress_data = self.ctrl_stress[ctrl_col].values
                con_nonstress_data = self.ctrl_nonstress[ctrl_col].values
                
                # 计算生长参数
                mut_stress_params = self.calculate_growth_parameters(mut_stress_data, self.time_points)
                mut_nonstress_params = self.calculate_growth_parameters(mut_nonstress_data, self.time_points)
                con_stress_params = self.calculate_growth_parameters(con_stress_data, self.time_points)
                con_nonstress_params = self.calculate_growth_parameters(con_nonstress_data, self.time_points)
                
                if any(params is None for params in [mut_stress_params, mut_nonstress_params, 
                                                   con_stress_params, con_nonstress_params]):
                    continue
                
                # 计算相对适应性 (Relative Fitness)
                # RF = (OD_mut_stress / OD_mut_nonstress) / (OD_con_stress / OD_con_nonstress)
                mut_ratio = mut_stress_params['final_od'] / mut_nonstress_params['final_od'] if mut_nonstress_params['final_od'] > 0 else 0
                con_ratio = con_stress_params['final_od'] / con_nonstress_params['final_od'] if con_nonstress_params['final_od'] > 0 else 0
                relative_fitness = mut_ratio / con_ratio if con_ratio > 0 else 0
                
                # 计算基于AUC的相对适应性
                mut_auc_ratio = mut_stress_params['auc'] / mut_nonstress_params['auc'] if mut_nonstress_params['auc'] > 0 else 0
                con_auc_ratio = con_stress_params['auc'] / con_nonstress_params['auc'] if con_nonstress_params['auc'] > 0 else 0
                relative_fitness_auc = mut_auc_ratio / con_auc_ratio if con_auc_ratio > 0 else 0
                
                # 计算生长速率的相对适应性
                mut_rate_ratio = mut_stress_params['max_growth_rate'] / mut_nonstress_params['max_growth_rate'] if mut_nonstress_params['max_growth_rate'] > 0 else 0
                con_rate_ratio = con_stress_params['max_growth_rate'] / con_nonstress_params['max_growth_rate'] if con_nonstress_params['max_growth_rate'] > 0 else 0
                relative_fitness_rate = mut_rate_ratio / con_rate_ratio if con_rate_ratio > 0 else 0
                
                # 计算滞后期比例
                mut_lag_ratio = mut_stress_params['lag_time'] / mut_nonstress_params['lag_time'] if mut_nonstress_params['lag_time'] > 0 else 1
                con_lag_ratio = con_stress_params['lag_time'] / con_nonstress_params['lag_time'] if con_nonstress_params['lag_time'] > 0 else 1
                relative_lag_ratio = mut_lag_ratio / con_lag_ratio if con_lag_ratio > 0 else 1
                
                # 综合相对适应性评分 (权重: 最终OD 40%, AUC 40%, 生长速率 20%)
                comprehensive_rf = (relative_fitness * 0.4 + relative_fitness_auc * 0.4 + relative_fitness_rate * 0.2)
                
                # 胁迫耐受性评分 (考虑滞后期惩罚)
                lag_penalty = 1 / max(relative_lag_ratio, 0.1) if relative_lag_ratio > 1 else 1
                tolerance_score = comprehensive_rf * lag_penalty * 100
                
                gene_results.append({
                    'gene': gene,
                    'mut_stress_final_od': mut_stress_params['final_od'],
                    'mut_nonstress_final_od': mut_nonstress_params['final_od'],
                    'con_stress_final_od': con_stress_params['final_od'],
                    'con_nonstress_final_od': con_nonstress_params['final_od'],
                    'mut_stress_auc': mut_stress_params['auc'],
                    'mut_nonstress_auc': mut_nonstress_params['auc'],
                    'con_stress_auc': con_stress_params['auc'],
                    'con_nonstress_auc': con_nonstress_params['auc'],
                    'mut_stress_growth_rate': mut_stress_params['max_growth_rate'],
                    'mut_nonstress_growth_rate': mut_nonstress_params['max_growth_rate'],
                    'con_stress_growth_rate': con_stress_params['max_growth_rate'],
                    'con_nonstress_growth_rate': con_nonstress_params['max_growth_rate'],
                    'relative_fitness_od': relative_fitness,
                    'relative_fitness_auc': relative_fitness_auc,
                    'relative_fitness_rate': relative_fitness_rate,
                    'comprehensive_rf': comprehensive_rf,
                    'relative_lag_ratio': relative_lag_ratio,
                    'tolerance_score': tolerance_score,
                    'control_column': ctrl_col
                })
                
            except Exception as e:
                print(f"处理基因 {gene} 时出错: {e}")
                continue
        
        self.results_df = pd.DataFrame(gene_results)
        print(f"成功分析了 {len(self.results_df)} 个基因")
        
    def filter_stress_tolerant_genes(self, rf_threshold_stress=1.05, comprehensive_rf_threshold_stress=1.05):
        """筛选胁迫耐受性基因（基于相对适应性）"""
        print(f"筛选胁迫耐受性基因 (综合相对适应性 > {comprehensive_rf_threshold_stress}, 最终OD相对适应性 > {rf_threshold_stress})...")
        
        # 应用筛选条件
        tolerant_genes = self.results_df[
            (self.results_df['comprehensive_rf'] > comprehensive_rf_threshold_stress) &
            (self.results_df['relative_fitness_od'] > rf_threshold_stress)
        ].copy()
        
        # 按综合相对适应性排序
        tolerant_genes = tolerant_genes.sort_values('comprehensive_rf', ascending=False)
        
        print(f"筛选出 {len(tolerant_genes)} 个胁迫耐受性基因")
        
        return tolerant_genes

    def filter_nonstress_tolerant_genes(self, rf_threshold_nonstress=0.95, comprehensive_rf_threshold_nonstress=0.95):
        """筛选胁迫不耐受基因：综合相对适应性 < threshold 且 最终OD相对适应性 < threshold"""
        print(f"筛选胁迫不耐受基因 (综合相对适应性 < {comprehensive_rf_threshold_nonstress}, 最终OD相对适应性 < {rf_threshold_nonstress})...")
        
        non_tolerant_genes = self.results_df[
            (self.results_df['comprehensive_rf'] < comprehensive_rf_threshold_nonstress) &
            (self.results_df['relative_fitness_od'] < rf_threshold_nonstress)
        ].copy()
        
        # 按综合相对适应性升序排序（不耐受更靠前）
        non_tolerant_genes = non_tolerant_genes.sort_values('comprehensive_rf', ascending=True)
        
        print(f"筛选出 {len(non_tolerant_genes)} 个胁迫不耐受基因")
        
        return non_tolerant_genes
    
    def generate_summary_statistics(self):
        """生成汇总统计"""
        print("生成汇总统计...")
        
        summary = {
            'total_genes': len(self.results_df),
            'relative_fitness_od_stats': self.results_df['relative_fitness_od'].describe(),
            'relative_fitness_auc_stats': self.results_df['relative_fitness_auc'].describe(),
            'comprehensive_rf_stats': self.results_df['comprehensive_rf'].describe(),
            'tolerance_score_stats': self.results_df['tolerance_score'].describe()
        }
        
        return summary
    
    def plot_tolerance_distribution(self, output_dir,rf_threshold_stress,comprehensive_rf_threshold_stress,rf_threshold_nonstress,comprehensive_rf_threshold_nonstress):
        """绘制耐受性指标分布图"""
        print("绘制耐受性指标分布图...")
        
        # 1) 相对适应性(OD)分布 -> PDF
        fig1, ax1 = plt.subplots(1, 1, figsize=(8, 6))
        ax1.hist(self.results_df['relative_fitness_od'], bins=50, alpha=0.7, color='skyblue', edgecolor='black')
        ax1.axvline(1.0, color='black', linestyle='-', alpha=0.5, label='Neutral (RF=1.0)')
        ax1.axvline(1.2, color='red', linestyle='--', label='Selection Threshold (1.2)')
        ax1.axvline(0.8, color='purple', linestyle='--', label='Non-tolerant Threshold (0.8)')
        ax1.set_xlabel('Relative Fitness OD')
        ax1.set_ylabel('Number of Genes')
        ax1.set_title('Relative Fitness Distribution OD')
        ax1.legend()
        fig1.tight_layout()
        fig1.savefig(output_dir / "relative_fitness_od_distribution.pdf", dpi=300, bbox_inches='tight')
        plt.close(fig1)
        
        # 2) 综合相对适应性分布 -> PDF
        fig2, ax2 = plt.subplots(1, 1, figsize=(8, 6))
        ax2.hist(self.results_df['comprehensive_rf'], bins=50, alpha=0.7, color='lightgreen', edgecolor='black')
        ax2.axvline(1.0, color='black', linestyle='-', alpha=0.5, label='Neutral (RF=1.0)')
        ax2.axvline(1.1, color='red', linestyle='--', label='Selection Threshold (1.1)')
        ax2.axvline(0.9, color='purple', linestyle='--', label='Non-tolerant Threshold (0.9)')
        ax2.set_xlabel('Comprehensive Relative Fitness')
        ax2.set_ylabel('Number of Genes')
        ax2.set_title('Comprehensive Relative Fitness Distribution')
        ax2.legend()
        fig2.tight_layout()
        fig2.savefig(output_dir / "comprehensive_rf_distribution.pdf", dpi=300, bbox_inches='tight')
        plt.close(fig2)
        
        # 3) 耐受性评分分布 -> PDF
        fig3, ax3 = plt.subplots(1, 1, figsize=(8, 6))
        ax3.hist(self.results_df['tolerance_score'], bins=50, alpha=0.7, color='orange', edgecolor='black')
        ax3.set_xlabel('Stress Tolerance Score')
        ax3.set_ylabel('Number of Genes')
        ax3.set_title('Stress Tolerance Score Distribution')
        fig3.tight_layout()
        fig3.savefig(output_dir / "tolerance_score_distribution.pdf", dpi=300, bbox_inches='tight')
        plt.close(fig3)
        
        # 4) 相对适应性散点图 -> PDF
        fig4, ax4 = plt.subplots(1, 1, figsize=(8, 6))
        scatter = ax4.scatter(self.results_df['relative_fitness_od'], self.results_df['comprehensive_rf'], 
                              c=self.results_df['tolerance_score'], cmap='viridis', alpha=0.6)
        ax4.axhline(rf_threshold_stress, color='red', linestyle='--', alpha=0.5)
        ax4.axvline(comprehensive_rf_threshold_stress, color='red', linestyle='--', alpha=0.5)
        ax4.axhline(1.0, color='black', linestyle='-', alpha=0.3)
        ax4.axvline(1.0, color='black', linestyle='-', alpha=0.3)
        ax4.axhline(rf_threshold_nonstress, color='purple', linestyle='--', alpha=0.5)
        ax4.axvline(comprehensive_rf_threshold_nonstress, color='purple', linestyle='--', alpha=0.5)
        ax4.set_xlabel('Relative Fitness OD')
        ax4.set_ylabel('Comprehensive Relative Fitness')
        ax4.set_title('Relative Fitness Comparison')
        plt.colorbar(scatter, ax=ax4, label='Stress Tolerance Score')
        fig4.tight_layout()
        fig4.savefig(output_dir / "rf_vs_comprehensive_scatter.pdf", dpi=300, bbox_inches='tight')
        plt.close(fig4)
        
    def save_results(self, output_dir,rf_threshold_stress,comprehensive_rf_threshold_stress,rf_threshold_nonstress,comprehensive_rf_threshold_nonstress):
        """保存分析结果"""
        output_dir = Path(output_dir)
        output_dir.mkdir(parents=True, exist_ok=True)
        
        print(f"保存结果到 {output_dir}")
        
        # 保存完整结果
        self.results_df.to_csv(output_dir / "all_genes_analysis.csv", index=False)
        
        # 筛选并保存胁迫耐受性基因
        tolerant_genes = self.filter_stress_tolerant_genes(rf_threshold_stress,comprehensive_rf_threshold_stress)
        tolerant_genes.to_csv(output_dir / "stress_tolerant_genes.csv", index=False)
        
        # 新增：筛选并保存胁迫不耐受基因
        non_tolerant_genes = self.filter_nonstress_tolerant_genes(rf_threshold_nonstress,comprehensive_rf_threshold_nonstress)
        non_tolerant_genes.to_csv(output_dir / "nonstress_tolerant_genes.csv", index=False)
        
        # 保存汇总统计
        summary = self.generate_summary_statistics()
        with open(output_dir / "summary_statistics.txt", 'w', encoding='utf-8') as f:
            f.write("胁迫耐受性分析汇总报告（基于相对适应性）\n")
            f.write("=" * 50 + "\n\n")
            f.write(f"总基因数: {summary['total_genes']}\n")
            f.write(f"胁迫耐受性基因数: {len(tolerant_genes)} ({len(tolerant_genes)/summary['total_genes']*100:.2f}%)\n\n")
            f.write(f"胁迫不耐受基因数: {len(non_tolerant_genes)} ({len(non_tolerant_genes)/summary['total_genes']*100:.2f}%)\n\n")
            
            f.write("相对适应性 (最终OD) 统计:\n")
            f.write(str(summary['relative_fitness_od_stats']) + "\n\n")
            
            f.write("相对适应性 (AUC) 统计:\n")
            f.write(str(summary['relative_fitness_auc_stats']) + "\n\n")
            
            f.write("综合相对适应性统计:\n")
            f.write(str(summary['comprehensive_rf_stats']) + "\n\n")
            
            f.write("胁迫耐受性评分统计:\n")
            f.write(str(summary['tolerance_score_stats']) + "\n\n")
            
            f.write(f"胁迫耐受性基因数量: {len(tolerant_genes)}\n")
            f.write(f"胁迫耐受性基因比例: {len(tolerant_genes)/summary['total_genes']*100:.2f}%\n")
        
        # 生成可视化图表
        self.plot_tolerance_distribution(output_dir,rf_threshold_stress,comprehensive_rf_threshold_stress,rf_threshold_nonstress,comprehensive_rf_threshold_nonstress)
        
        print("结果保存完成!")
        return tolerant_genes

def main():
    """主函数"""
    import argparse
    parser = argparse.ArgumentParser(description='处理高通量生长曲线数据')
    parser.add_argument('--input_dir', type=str, required=True,
                        help='输入数据目录(02.cleaned_data的父目录)')
    parser.add_argument('--output_dir', type=str, required=True,
                        help='输出数据目录(03.stress_tolerance_analysis)')
    parser.add_argument('--rf_threshold_stress', type=float, default=1.05,
                        help='胁迫最终OD相对适应性阈值,默认1.05')
    parser.add_argument('--comprehensive_rf_threshold_stress', type=float, default=1.05,
                        help='胁迫综合相对适应性阈值,默认1.05')
    parser.add_argument('--rf_threshold_nonstress', type=float, default=0.95,
                        help='非胁迫最终OD相对适应性阈值,默认0.95')
    parser.add_argument('--comprehensive_rf_threshold_nonstress', type=float, default=0.95,
                        help='非胁迫综合相对适应性阈值,默认0.95')
    args = parser.parse_args()

    # 创建分析器
    analyzer = StressToleranceAnalyzer(
        args.input_dir,
        args.rf_threshold_stress,
        args.comprehensive_rf_threshold_stress,
        args.rf_threshold_nonstress,
        args.comprehensive_rf_threshold_nonstress,
    )
    
    # 执行分析
    analyzer.load_data()
    analyzer.analyze_gene_performance()
    tolerant_genes = analyzer.save_results(
        args.output_dir,
        args.rf_threshold_stress,
        args.comprehensive_rf_threshold_stress,
        args.rf_threshold_nonstress,
        args.comprehensive_rf_threshold_nonstress
    )
    
    # 显示前10个最佳胁迫耐受性基因
    print("\n" + "="*60)
    print("前10个最佳胁迫耐受性基因:")
    print("="*60)
    
    if len(tolerant_genes) > 0:
        top_genes = tolerant_genes.head(10)
        for i, (_, gene_data) in enumerate(top_genes.iterrows(), 1):
            print(f"{i:2d}. {gene_data['gene']}")
            print(f"    综合评分: {gene_data['tolerance_score']:.2f}")
            print(f"    综合相对适应性: {gene_data['comprehensive_rf']:.3f}")
            print(f"    相对适应性(OD): {gene_data['relative_fitness_od']:.3f}")
            print(f"    胁迫条件最终OD: {gene_data['mut_stress_final_od']:.3f}")
            print(f"    非胁迫条件最终OD: {gene_data['mut_nonstress_final_od']:.3f}")
            print()
    else:
        print("未找到符合条件的胁迫耐受性基因，建议调整筛选阈值")
    
    print(f"详细结果已保存到: {args.output_dir}")

    

if __name__ == "__main__":
    main()
