import pandas as pd
import seaborn as sns
import matplotlib.pyplot as plt
from pathlib import Path
import numpy as np

def plot_feature_correlation_scatter(input_file, output_dir, feature1, feature2):
    """
    绘制两个特征在所有基因、所有条件下的散点图，并计算总体相关系数。
    """
    df = pd.read_csv(input_file)
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    
    # 提取所有条件下的这两列数据
    conditions = ['mut_stress', 'mut_nonstress', 'con_stress', 'con_nonstress']
    
    data_points = []
    
    for _, row in df.iterrows():
        gene = row['gene']
        for cond in conditions:
            try:
                val1 = float(row[f'{cond}_{feature1}'])
                val2 = float(row[f'{cond}_{feature2}'])
                
                # 过滤掉 NaN 和 无穷大
                if np.isfinite(val1) and np.isfinite(val2):
                    data_points.append({
                        'gene': gene,
                        'condition': cond,
                        feature1: val1,
                        feature2: val2
                    })
            except (KeyError, ValueError):
                continue
                
    plot_df = pd.DataFrame(data_points)
    
    if plot_df.empty:
        print(f"No valid data found for {feature1} vs {feature2}")
        return

    # 计算总体相关系数
    overall_corr = plot_df[[feature1, feature2]].corr().iloc[0, 1]
    
    # 绘图
    plt.figure(figsize=(10, 8))
    
    # 使用不同颜色表示不同条件
    sns.scatterplot(data=plot_df, x=feature1, y=feature2, hue='condition', alpha=0.6, s=60)
    
    # 添加回归线 (基于所有数据)
    sns.regplot(data=plot_df, x=feature1, y=feature2, scatter=False, color='black', line_kws={'linestyle': '--'})
    
    plt.title(f'Feature Correlation: {feature1} vs {feature2}\nOverall Pearson r = {overall_corr:.4f} (n={len(plot_df)})')
    plt.grid(True, alpha=0.3)
    
    # 保存
    safe_f1 = feature1.replace('/', '_').replace(' ', '_')
    safe_f2 = feature2.replace('/', '_').replace(' ', '_')
    output_path = output_dir / f'scatter_all_conditions_{safe_f1}_vs_{safe_f2}.png'
    plt.savefig(output_path)
    print(f"Plot saved to: {output_path}")
    print(f"Correlation between {feature1} and {feature2}: {overall_corr:.4f}")

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument('--input', type=str, required=True, help='Path to all_features_raw_50plus.csv')
    parser.add_argument('--output_dir', type=str, required=True)
    parser.add_argument('--feat1', type=str, required=True)
    parser.add_argument('--feat2', type=str, required=True)
    args = parser.parse_args()
    
    plot_feature_correlation_scatter(args.input, args.output_dir, args.feat1, args.feat2)
