import pandas as pd
import numpy as np
from pathlib import Path
import argparse

# Features where lower values are better (or negative values closer to 0 are better)
# For these, we will invert the RF (1/RF) so that higher RF score means better performance.
LOWER_IS_BETTER = [
    'lag_phase', 't_detect', 't_lag_end', 
    'decline_fraction', 'max_decline_rate', 
    't10', 't50', 't90', 
    'time_max_OD', 't_peak_from_start', 
    'Lag_cost_index', 
    'AIC', 'BIC', 'fit_RMSE',
    'plateau_mean_OD' # Wait, plateau mean OD usually higher is better?
]

# Correcting list based on biological meaning
# plateau_mean_OD: Higher biomass is usually better. So remove from list.
LOWER_IS_BETTER = [
    'lag_phase', 't_detect', 't_lag_end', 
    'decline_fraction', 
    't10', 't50', 't90', 
    'time_max_OD', 't_peak_from_start', 
    'Lag_cost_index', 
    'AIC', 'BIC', 'fit_RMSE',
    'post_plateau_noise' # Lower noise is better
]

# Note: max_decline_rate is typically negative.
# If Mut=-0.1 (good), Con=-0.5 (bad). Ratio = 0.2. 1/Ratio = 5.
# If Mut=-0.5 (bad), Con=-0.1 (good). Ratio = 5. 1/Ratio = 0.2.
# So 1/Ratio works for max_decline_rate too.
LOWER_IS_BETTER.append('max_decline_rate')


def calculate_rf(row, feature):
    """
    Calculate Relative Fitness (RF) for a single feature.
    RF = (Mut_Stress / Mut_NonStress) / (Con_Stress / Con_NonStress)
    """
    try:
        # Get values
        ms = float(row.get(f'mut_stress_{feature}', np.nan))
        mn = float(row.get(f'mut_nonstress_{feature}', np.nan))
        cs = float(row.get(f'con_stress_{feature}', np.nan))
        cn = float(row.get(f'con_nonstress_{feature}', np.nan))
        
        # Handle zeros and NaNs
        # If any value is NaN, return NaN
        if np.isnan([ms, mn, cs, cn]).any():
            return np.nan
            
        # Avoid division by zero
        epsilon = 1e-6
        if abs(mn) < epsilon: mn = epsilon
        if abs(cs) < epsilon: cs = epsilon
        if abs(cn) < epsilon: cn = epsilon
        
        # Calculate Ratios
        ratio_mut = ms / mn
        ratio_con = cs / cn
        
        if abs(ratio_con) < epsilon:
            return np.nan
            
        rf = ratio_mut / ratio_con
        
        # Invert if lower is better
        if feature in LOWER_IS_BETTER:
            if abs(rf) < epsilon:
                rf = np.nan # Avoid 1/0
            else:
                rf = 1.0 / rf
                
        return rf
        
    except Exception as e:
        return np.nan

def main():
    parser = argparse.ArgumentParser(description='Calculate RF Matrix for Selected Features')
    parser.add_argument('--input_features', required=True, help='Path to all_features_raw_50plus.csv')
    parser.add_argument('--input_selected', required=True, help='Path to selected_features_list.csv')
    parser.add_argument('--output_dir', required=True, help='Output directory')
    
    args = parser.parse_args()
    
    # Load data
    print(f"Loading features from {args.input_features}...")
    df_features = pd.read_csv(args.input_features)
    
    print(f"Loading selected features list from {args.input_selected}...")
    df_selected = pd.read_csv(args.input_selected)
    selected_features = df_selected['Feature'].tolist()
    
    print(f"Calculating RF for {len(selected_features)} features...")
    
    # Prepare results
    gene_list = []
    # Extract gene names from columns if not explicit, usually 'gene' column exists
    if 'gene' in df_features.columns:
        gene_list = df_features['gene'].tolist()
    else:
        # Try to infer or fail
        print("Error: 'gene' column not found in input features.")
        return

    rf_data = {}
    
    for feature in selected_features:
        # Check if feature columns exist
        if f'mut_stress_{feature}' not in df_features.columns:
            print(f"Warning: Feature {feature} columns not found. Skipping.")
            continue
            
        # Calculate RF for all genes
        rf_values = []
        for _, row in df_features.iterrows():
            rf = calculate_rf(row, feature)
            rf_values.append(rf)
            
        rf_data[feature] = rf_values
        
    # Create DataFrame
    df_rf = pd.DataFrame(rf_data, index=gene_list)
    
    # Drop rows where all RFs are NaN (failed calculations)
    df_rf = df_rf.dropna(how='all')
    
    # Save
    output_dir = Path(args.output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    output_csv = output_dir / 'rf_matrix_final.csv'
    
    df_rf.to_csv(output_csv)
    print(f"RF Matrix saved to: {output_csv}")
    print(f"Matrix shape: {df_rf.shape}")

if __name__ == "__main__":
    main()
