import pandas as pd
import numpy as np
from pathlib import Path

def screen_top_strains(rf_matrix_file, output_dir, top_n=5):
    # 读取 RF 矩阵 (index=gene)
    df = pd.read_csv(rf_matrix_file, index_col=0)
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    
    # 存储结果的列表
    results = []
    
    # 记录所有上榜的基因 (用于后续画图)
    all_top_genes = set()
    
    # 遍历每一列 (每一个特征)
    for feature in df.columns:
        # 排除非数值列 (如果有)
        if df[feature].dtype == 'object':
            continue
        
        top_genes = df[feature].nlargest(top_n)
        
        for rank, (gene, rf_val) in enumerate(top_genes.items(), 1):
            results.append({
                'Feature': feature,
                'Rank': rank,
                'Gene': gene,
                'RF_Value': rf_val
            })
            all_top_genes.add(gene)
            
    # 转换为 DataFrame 并保存
    results_df = pd.DataFrame(results)
    output_csv = output_dir / 'top5_strains_per_feature.csv'
    results_df.to_csv(output_csv, index=False)
    
    # 保存去重后的基因列表
    gene_list_file = output_dir / 'unique_top_genes_list.txt'
    with open(gene_list_file, 'w') as f:
        for gene in sorted(list(all_top_genes)):
            f.write(f"{gene}\n")
            
    print(f"Screening complete.")
    print(f"Total unique genes selected: {len(all_top_genes)}")
    print(f"Results saved to: {output_csv}")
    print(f"Gene list saved to: {gene_list_file}")
    
    # 打印一些摘要
    print("\nTop 1 Strains for key features:")
    key_feats = ['max_OD', 'max_mu', 'AUC', 'decline_fraction', 'lag_phase']
    for feat in key_feats:
        if feat in df.columns:
            top1 = df[feat].idxmax()
            val = df[feat].max()
            print(f"  {feat}: {top1} (RF={val:.2f})")

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument('--input', type=str, required=True, help='Path to rf_matrix_final.csv')
    parser.add_argument('--output_dir', type=str, required=True)
    parser.add_argument('--top_n', type=int, default=5)
    args = parser.parse_args()
    
    screen_top_strains(args.input, args.output_dir, args.top_n)
