```python

import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from pathlib import Path

# 严格按照 Excel 表格定义的分类 (合并了部分极小的类以避免颜色过多)
FEATURE_CLASSES_EXACT = {
    'Model fit and quality control': [
        'AIC / BIC', 'fit_RMSE', 'AIC', 'BIC'
    ],
    'Integrated growth output': [
        'AUC', 'AUC_early', 'AUC_late', 'AUC_mid', 
        'back_loaded_fraction', 'front_loaded_fraction'
    ],
    'Decline and survival': [
        'decline_fraction', 'decline_onset_time', 'half_life_after_peak', 
        'late_OD_slope', 'max_decline_rate', 'persistence_time'
    ],
    'Multi-phase / Diauxic growth': [
        'diauxic_lag', 'n_phases', 'second_mu_max'
    ],
    'Biomass yield and carrying capacity': [
        'Endpoint_biomass', 'max_OD', 'model_A', 'model_A (asymptote)'
    ],
    'Exponential phase properties': [
        'exp_fit_R2', 'exp_OD_gain', 'Exp. phase length', 'Exp_phase_length',
        'mean_mu', 't_exp_end'
    ],
    'Growth curve shape and dynamics': [
        'growth_timing_asymmetry', 'max_acceleration', 'max_deceleration', 
        'model_shape', 'model_shape (ν, etc.)', 't_inflection'
    ],
    'Intrinsic growth capacity and kinetics': [
        'initial_mu', 'initial_mu (μ₀)', 'max_dODdt', 'max_mu', 'mean_mu'
    ],
    'Lag and growth inhibition': [
        'Lag_cost_index', 'lag_phase', 't_detect', 't_lag_end'
    ],
    'Growth timing and progression': [
        'OD_centroid_time', 't_AUC50', 't_peak_from_start', 
        't10', 't50', 't90', 'time_max_mu', 
        'time_max_OD', 'time_max_OD / t_peak', 'OD_time_skewness'
    ],
    'Plateau and stationary phase properties': [
        'plateau_mean_OD', 'post_plateau_noise', 'stationary_duration', 't_plateau_start'
    ]
}

def get_feature_class_exact(feature_name):
    # 1. 尝试完全匹配
    for cls, keywords in FEATURE_CLASSES_EXACT.items():
        if feature_name in keywords:
            return cls
            
    # 2. 尝试部分匹配 (处理可能的命名差异)
    for cls, keywords in FEATURE_CLASSES_EXACT.items():
        for k in keywords:
            # 简单的包含关系检查
            if k == feature_name or (len(k) > 3 and k in feature_name):
                return cls
                
    return 'Other'

def filter_features(corr_matrix, threshold=0.9):
    """
    根据相关性矩阵筛选特征
    """
    to_drop = set()
    # 优先保留的特征列表 (更基础、更好解释)
    priority_features = [
        'max_OD', 'AUC', 'max_mu', 'lag_phase', 'decline_fraction', 
        't50', 'growth_timing_asymmetry'
    ]
    
    columns = corr_matrix.columns.tolist()
    
    for i in range(len(columns)):
        col_i = columns[i]
        if col_i in to_drop:
            continue
            
        for j in range(i + 1, len(columns)):
            col_j = columns[j]
            if col_j in to_drop:
                continue
                
            if abs(corr_matrix.iloc[i, j]) > threshold:
                # 冲突！决定删谁
                if col_i in priority_features and col_j not in priority_features:
                    to_drop.add(col_j)
                elif col_j in priority_features and col_i not in priority_features:
                    to_drop.add(col_i)
                else:
                    # 默认保留名字短的，通常更简洁
                    if len(col_i) < len(col_j):
                        to_drop.add(col_j)
                    else:
                        to_drop.add(col_i)
                        
    selected_features = [c for c in columns if c not in to_drop]
    return selected_features

def plot_raw_correlation_heatmap(input_csv, output_dir):
    df = pd.read_csv(input_csv)
    output_dir = Path(output_dir)
    
    # 1. 构建长格式数据
    feature_names = set()
    for col in df.columns:
        for prefix in ['mut_stress_', 'mut_nonstress_', 'con_stress_', 'con_nonstress_']:
            if col.startswith(prefix):
                feature_names.add(col.replace(prefix, ''))
                break
    feature_list = sorted(list(feature_names))
    
    combined_data = []
    conditions = ['mut_stress', 'mut_nonstress', 'con_stress', 'con_nonstress']
    
    for _, row in df.iterrows():
        for cond in conditions:
            row_dict = {}
            for feat in feature_list:
                col_name = f"{cond}_{feat}"
                if col_name in df.columns:
                    row_dict[feat] = row[col_name]
                else:
                    row_dict[feat] = np.nan
            combined_data.append(row_dict)
            
    long_df = pd.DataFrame(combined_data)
    
    # 清洗数据
    missing = long_df.isnull().mean()
    long_df = long_df.loc[:, missing < 0.2]
    long_df = long_df.fillna(long_df.median())
    variances = long_df.var()
    long_df = long_df.loc[:, variances > 1e-6]
    
    # 2. 计算相关矩阵
    corr_matrix = long_df.corr()
    
    # *** 新增：在此处进行特征筛选 ***
    selected_feats_list = filter_features(corr_matrix)
    selected_feats = set(selected_feats_list)
    print(f"Selected {len(selected_feats)} independent features from {len(corr_matrix.columns)}.")
    
    # 保存筛选结果
    pd.DataFrame(selected_feats_list, columns=['Feature']).to_csv(output_dir / 'selected_features_list.csv', index=False)
    
    # *** 新增：保存相关性矩阵 ***
    corr_matrix.to_csv(output_dir / 'raw_feature_correlation_matrix.csv')
    print(f"Correlation matrix saved to: {output_dir / 'raw_feature_correlation_matrix.csv'}")
    
    # 3. 排序 (按 Exact Class)
    sorted_cols = sorted(corr_matrix.columns, key=lambda x: (
        list(FEATURE_CLASSES_EXACT.keys()).index(get_feature_class_exact(x)) 
        if get_feature_class_exact(x) in FEATURE_CLASSES_EXACT else 99, 
        x
    ))
    corr_sorted = corr_matrix.loc[sorted_cols, sorted_cols]
    
    # 4. 绘制热图
    plt.figure(figsize=(20, 18)) # 增大画布尺寸
    
    # 颜色条标记类别
    unique_classes = list(FEATURE_CLASSES_EXACT.keys()) + ['Other']
    # 使用tab20调色板以支持更多分类
    class_colors = sns.color_palette("tab20", len(unique_classes))
    class_map = {c: color for c, color in zip(unique_classes, class_colors)}
    
    row_colors = [class_map[get_feature_class_exact(col)] for col in sorted_cols]
    
    # 使用刚刚筛选出的特征进行标记
    xticklabels = [f"★ {col}" if col in selected_feats else col for col in sorted_cols]
    yticklabels = xticklabels

    # 绘制 ClusterMap (关闭聚类，仅用作带颜色条的热图)
    # *** 新增：添加数字标注 (annot=True) ***
    # fmt='.2f' 保留两位小数
    # annot_kws={'size': 6} 设置字体大小，因为格子多，字体要小
    g = sns.clustermap(corr_sorted, 
                       cmap='coolwarm', center=0, vmin=-1, vmax=1,
                       row_cluster=False, col_cluster=False,
                       row_colors=row_colors, col_colors=row_colors,
                       figsize=(24, 22), linewidths=.5, # 再增大一点尺寸以便容纳数字
                       xticklabels=xticklabels, yticklabels=yticklabels,
                       dendrogram_ratio=(0.01, 0.01),
                       cbar_pos=(0.02, 0.8, 0.01, 0.15),
                       annot=True, fmt='.2f', annot_kws={'size': 6})
    
    # 调整热图位置，留出右侧给图例
    # [left, bottom, width, height]
    # 缩小 width 从 0.65 到 0.6，给右侧留更多空间
    g.ax_heatmap.set_position([0.1, 0.1, 0.6, 0.8])
    g.ax_row_colors.set_position([0.07, 0.1, 0.03, 0.8])
    g.ax_col_colors.set_position([0.1, 0.9, 0.6, 0.03])
    
    # 添加图例 (放在右侧外部)
    from matplotlib.patches import Patch
    handles = [Patch(facecolor=class_map[c], label=c) for c in unique_classes 
               if any(get_feature_class_exact(f) == c for f in sorted_cols)]
    
    # 在 Figure 对象上添加图例
    # bbox_to_anchor=(0.8, 0.8) 相对于整个figure的坐标，向右移动以避免遮挡列名
    g.fig.legend(handles=handles, title=' ', 
                 bbox_to_anchor=(0.8, 0.9), loc='upper left', 
                 borderaxespad=0., fontsize=12, title_fontsize=14,
                 frameon=False,  # 去掉边框
                 facecolor='none',  # 完全透明背景
                )
    
    g.fig.suptitle('Raw Feature Correlation Matrix \n(★ = Selected Features)', 
                   fontsize=20, y=0.98)
    
    output_path = output_dir / 'raw_feature_correlation_heatmap.png'
    plt.savefig(output_path)
    print(f"Heatmap saved to: {output_path}")

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument('--input', type=str, required=True)
    parser.add_argument('--output_dir', type=str, required=True)
    args = parser.parse_args()
    
    plot_raw_correlation_heatmap(args.input, args.output_dir)
```
