import numpy as np
from scipy.optimize import curve_fit
from scipy.signal import savgol_filter
from pathlib import Path
import warnings
import sys

warnings.filterwarnings('ignore')

# ---- Constants from calcFeat_Glc.py ----
OD_EPS = 1e-6
MU_WINDOW = 11
MU_R2_MIN = 0.9
MU_BASELINE_N = 5
MU_OD_THRESHOLD_ABS = 0.01
MU_T_LAG_MIN = 2.5
MU_CAP_PER_H = 2.0

# ---- Helper Functions ----

def _safe_trapz(y: np.ndarray, t: np.ndarray) -> float:
    if len(t) < 2 or not (np.all(np.isfinite(t)) and np.all(np.isfinite(y))):
        return np.nan
    try:
        return float(np.trapz(y, t))
    except AttributeError:
        # NumPy 2.0+
        return float(np.trapezoid(y, t))

def _shannon_entropy(p: np.ndarray) -> float:
    p = np.asarray(p, dtype=float)
    p = p[np.isfinite(p)]
    s = np.sum(p)
    if s <= 0:
        return np.nan
    p = p / s
    p = p[p > 0]
    return float(-np.sum(p * np.log(p))) if p.size else np.nan

def _weighted_time_skewness(t: np.ndarray, w: np.ndarray) -> float:
    t = np.asarray(t, dtype=float)
    w = np.asarray(w, dtype=float)
    mask = np.isfinite(t) & np.isfinite(w) & (w >= 0)
    if mask.sum() < 3:
        return np.nan
    t = t[mask]; w = w[mask]
    s = float(np.sum(w))
    if s <= 0:
        return np.nan
    w = w / s
    mu = float(np.sum(w * t))
    v = float(np.sum(w * (t - mu) ** 2))
    if v <= 0 or not np.isfinite(v):
        return np.nan
    m3 = float(np.sum(w * (t - mu) ** 3))
    return float(m3 / (v ** 1.5))

def _detect_plateau(t: np.ndarray, y: np.ndarray, slope_eps: Optional[float] = None, window: int = 5) -> tuple:
    if len(t) < max(3, window + 1):
        return np.nan, np.zeros(len(t), dtype=bool)

    dt = np.diff(t)
    dy = np.diff(y)
    with np.errstate(divide='ignore', invalid='ignore'):
        slopes = np.where(dt > 0, dy / dt, np.nan)

    max_slope = np.nanmax(np.abs(slopes)) if np.any(np.isfinite(slopes)) else np.nan
    if slope_eps is None:
        slope_eps = max(0.02 * max_slope, 1e-4) if np.isfinite(max_slope) else 1e-4

    i_start = int(0.3 * (len(t) - 1))
    ok = np.abs(slopes) <= slope_eps
    ok[:i_start] = False

    for i in range(i_start, len(ok) - window + 1):
        if np.all(ok[i : i + window]):
            plateau_mask = np.zeros(len(t), dtype=bool)
            plateau_mask[i:] = True
            return float(t[i]), plateau_mask

    i = int(0.8 * len(t))
    plateau_mask = np.zeros(len(t), dtype=bool)
    plateau_mask[i:] = True
    return float(t[i]), plateau_mask

def _time_to_level(y: np.ndarray, t: np.ndarray, level: float) -> float:
    if not np.isfinite(level):
        return np.nan
    if np.nanmax(y) < level:
        return np.nan
    idx = np.searchsorted(y, level)
    if 0 < idx < len(y):
        return float(np.interp(level, y[idx - 1 : idx + 1], t[idx - 1 : idx + 1]))
    if idx == 0:
        return float(t[0])
    return float(t[-1])

def _preprocess_od_for_mu(y: np.ndarray) -> np.ndarray:
    y = np.maximum(np.asarray(y, dtype=float), OD_EPS)
    try:
        if len(y) >= 5:
            w = min(5, len(y) if len(y) % 2 == 1 else len(y) - 1)
            y = savgol_filter(y, window_length=w, polyorder=2)
        elif len(y) >= 3:
            m = np.median(np.stack([y[:-2], y[1:-1], y[2:]], axis=1), axis=1)
            y = np.concatenate([[y[0]], m, [y[-1]]])
    except:
        pass
    return np.maximum(y, OD_EPS)

def _specific_growth_rate_mu_t(t: np.ndarray, y: np.ndarray) -> tuple:
    w = MU_WINDOW
    if len(t) < w or len(y) < w:
        return np.array([]), np.array([]), np.array([]), np.array([])
    t = np.asarray(t, dtype=float)
    y = np.asarray(y, dtype=float)
    y_smooth = _preprocess_od_for_mu(y)
    log_od = np.log(np.maximum(y_smooth, OD_EPS))
    
    try:
        if len(log_od) >= 5:
            w_sg = min(w, len(log_od) if len(log_od) % 2 == 1 else len(log_od) - 1)
            if w_sg >= 3:
                log_od_smooth = savgol_filter(log_od, window_length=w_sg, polyorder=2)
            else:
                log_od_smooth = log_od
        else:
            log_od_smooth = log_od
    except:
        log_od_smooth = log_od

    mu = np.gradient(log_od_smooth, t)
    mu = np.where(np.isfinite(mu) & (mu > MU_CAP_PER_H), MU_CAP_PER_H, mu)
    
    r2_dummy = np.full_like(t, np.nan)
    od_min_dummy = np.full_like(t, np.nanmin(y_smooth))
    return t.copy(), mu, r2_dummy, od_min_dummy

def _exp_phase_bounds_from_mu(t_c: np.ndarray, mu: np.ndarray, good: np.ndarray) -> tuple:
    if len(t_c) == 0 or not np.any(good):
        return np.nan, np.nan
    tt = t_c[good]
    return float(np.nanmin(tt)), float(np.nanmax(tt))

def _local_maxima(x: np.ndarray) -> np.ndarray:
    x = np.asarray(x, dtype=float)
    if x.size < 3:
        return np.array([], dtype=int)
    return np.where((x[1:-1] > x[:-2]) & (x[1:-1] >= x[2:]) & np.isfinite(x[1:-1]))[0] + 1

def _fit_gompertz(t: np.ndarray, y: np.ndarray) -> dict:
    try:
        if len(t) < 6 or not np.all(np.isfinite(t)) or not np.all(np.isfinite(y)):
            return {"model_A": np.nan, "model_shape": np.nan, "fit_RMSE": np.nan, "AIC": np.nan, "BIC": np.nan}

        def gompertz(tt, A, B, C, y0):
            return A * np.exp(-np.exp(B - C * tt)) + y0

        y0_guess = float(np.nanmin(y))
        A_guess = float(np.nanmax(y) - y0_guess)
        C_guess = 1.0 / (np.nanmax(t) - np.nanmin(t) + 1e-9)
        B_guess = 1.0

        bounds = ([0, -np.inf, 0, -np.inf], [np.inf, np.inf, np.inf, np.inf])
        popt, _ = curve_fit(gompertz, t, y, p0=[A_guess, B_guess, C_guess, y0_guess], bounds=bounds, maxfev=20000)
        
        yhat = gompertz(t, *popt)
        resid = y - yhat
        rmse = float(np.sqrt(np.nanmean(resid**2)))
        rss = float(np.nansum(resid**2))
        n = int(np.isfinite(y).sum())
        k = 4
        if n <= k or rss <= 0:
            aic = bic = np.nan
        else:
            aic = float(n * np.log(rss / n) + 2 * k)
            bic = float(n * np.log(rss / n) + k * np.log(n))
        
        return {
            "model_A": float(popt[0] + popt[3]),
            "model_shape": float(popt[2]),
            "fit_RMSE": rmse,
            "AIC": aic,
            "BIC": bic,
        }
    except Exception:
        return {"model_A": np.nan, "model_shape": np.nan, "fit_RMSE": np.nan, "AIC": np.nan, "BIC": np.nan}

def calc_curve_features(time: np.ndarray, od: np.ndarray) -> dict:
    """Compute all 50+ growth-curve features."""
    mask = np.isfinite(od) & np.isfinite(time)
    if mask.sum() < 3:
        return {}

    t = np.asarray(time, dtype=float)[mask]
    y = np.asarray(od, dtype=float)[mask]
    order = np.argsort(t)
    t, y = t[order], y[order]

    nan = np.nan
    t_min, t_max = float(t[0]), float(t[-1])
    duration_h = float(t_max - t_min) if t_max > t_min else nan

    # Basic stats
    max_OD = float(np.nanmax(y))
    min_OD = float(np.nanmin(y))
    Endpoint_biomass = float(y[-1])
    AUC = _safe_trapz(y, t)

    # dOD/dt
    dt = np.diff(t)
    dy = np.diff(y)
    with np.errstate(divide='ignore', invalid='ignore'):
        slopes = np.where(dt > 0, dy / dt, np.nan)
    max_dODdt = float(np.nanmax(slopes)) if np.any(np.isfinite(slopes)) else nan

    # Mu (growth rate)
    t_c, mu_w, r2_w, od_min_w = _specific_growth_rate_mu_t(t, y)
    
    if len(mu_w):
        baseline_n = min(MU_BASELINE_N, max(2, len(y) // 3))
        od0 = float(np.nanmean(y[:baseline_n]))
        sd0 = float(np.nanstd(y[:baseline_n]))
        sd0 = sd0 if np.isfinite(sd0) and sd0 > 0 else 0.0
        detect_thresh = max(od0 + 3 * sd0, MU_OD_THRESHOLD_ABS)
        good_mu = (t_c >= MU_T_LAG_MIN) & (od_min_w >= detect_thresh) & np.isfinite(mu_w)
        if not np.any(good_mu):
            good_mu = (t_c >= MU_T_LAG_MIN) & np.isfinite(mu_w)
        if not np.any(good_mu):
            good_mu = np.isfinite(mu_w)
    else:
        good_mu = np.array([], dtype=bool)
        od0 = float(np.nanmean(y[: min(MU_BASELINE_N, len(y))]))
        sd0 = float(np.nanstd(y[: min(MU_BASELINE_N, len(y))]))
        detect_thresh = max(od0 + 3 * (sd0 if np.isfinite(sd0) else 0.0), MU_OD_THRESHOLD_ABS)

    max_mu = float(np.nanmax(mu_w[good_mu])) if len(mu_w) and np.any(good_mu) else nan
    mean_mu = float(np.nanmean(mu_w[good_mu])) if len(mu_w) and np.any(good_mu) else nan
    doubling_time_min = float(np.log(2) / max_mu * 60) if np.isfinite(max_mu) and max_mu > 0 else nan
    time_max_mu = float(t_c[good_mu][np.nanargmax(mu_w[good_mu])]) if len(mu_w) and np.any(good_mu) else nan
    OD_max_mu = nan
    if np.isfinite(time_max_mu) and np.nanmin(t) <= time_max_mu <= np.nanmax(t):
        OD_max_mu = float(np.interp(time_max_mu, t, y))

    exp_start, exp_end = _exp_phase_bounds_from_mu(t_c, mu_w, good_mu) if len(mu_w) and np.any(good_mu) else (nan, nan)
    t_exp_end = exp_end
    Exp_phase_length = float(exp_end - exp_start) if np.isfinite(exp_start) and np.isfinite(exp_end) and exp_end > exp_start else nan

    time_max_OD = float(t[np.nanargmax(y)])
    t_peak = time_max_OD
    time_to_plateau, plateau_mask = _detect_plateau(t, y)
    plateau_vals = y[plateau_mask] if np.any(plateau_mask) else np.array([])
    plateau_mean = float(np.nanmean(plateau_vals)) if plateau_vals.size else nan
    plateau_variability = float(np.nanstd(plateau_vals)) if plateau_vals.size else nan
    post_plateau_noise = float(np.nanstd(y[-max(3, len(y)//10):])) if len(y) >= 3 else nan

    t_detect = _time_to_level(y, t, detect_thresh)
    t_lag_end = t_detect
    lag_phase = float(t_detect - t_min) if np.isfinite(t_detect) else nan

    net = plateau_mean - od0 if np.isfinite(plateau_mean) else nan
    if np.isfinite(net) and net > 0:
        t10 = _time_to_level(y, t, od0 + 0.10 * net)
        t50 = _time_to_level(y, t, od0 + 0.50 * net)
        t90 = _time_to_level(y, t, od0 + 0.90 * net)
    else:
        t10 = t50 = t90 = nan

    growth_timing_asymmetry = float((t90 - t50) / (t50 - t10)) if all(np.isfinite([t10, t50, t90])) and (t50 - t10) != 0 else nan

    if np.isfinite(duration_h) and duration_h > 0:
        t1 = t_min + duration_h / 3
        t2 = t_min + 2 * duration_h / 3
        i1 = np.searchsorted(t, t1)
        i2 = np.searchsorted(t, t2)
        i1 = int(np.clip(i1, 1, len(t) - 1))
        i2 = int(np.clip(i2, i1 + 1, len(t)))
        AUC_early = _safe_trapz(y[:i1], t[:i1])
        AUC_mid = _safe_trapz(y[i1-1:i2], t[i1-1:i2])
        AUC_late = _safe_trapz(y[i2-1:], t[i2-1:])
    else:
        AUC_early = AUC_mid = AUC_late = nan

    front_loaded_fraction = float(AUC_early / AUC) if np.isfinite(AUC_early) and np.isfinite(AUC) and AUC != 0 else nan
    back_loaded_fraction = float(AUC_late / AUC) if np.isfinite(AUC_late) and np.isfinite(AUC) and AUC != 0 else nan

    t_AUC50 = nan
    if len(t) >= 3 and np.isfinite(AUC):
        inc = (y[1:] + y[:-1]) / 2 * np.diff(t)
        cum = np.concatenate([[0.0], np.cumsum(inc)])
        half = 0.5 * cum[-1]
        j = np.searchsorted(cum, half)
        if 0 < j < len(t):
            t_AUC50 = float(np.interp(half, [cum[j-1], cum[j]], [t[j-1], t[j]]))

    max_acceleration = max_deceleration = t_inflection = nan
    if len(t) >= 5:
        yy = _preprocess_od_for_mu(y)
        dy1 = np.gradient(yy, t)
        dy2 = np.gradient(dy1, t)
        if np.any(np.isfinite(dy2)):
            max_acceleration = float(np.nanmax(dy2))
            max_deceleration = float(np.nanmin(dy2))
            sign = np.sign(dy2)
            zc = np.where(np.diff(sign) != 0)[0]
            if zc.size:
                t_inflection = float(t[zc[0]])
            else:
                t_inflection = float(t[np.nanargmax(dy2)])

    decline_onset_time = max_decline_rate = half_life_after_peak = decline_fraction = late_OD_slope = persistence_time = nan
    if np.isfinite(t_peak):
        half_level = 0.5 * max_OD
        if np.nanmin(y[t >= t_peak]) <= half_level:
            mask_after = t >= t_peak
            yy = y[mask_after]
            tt = t[mask_after]
            below = np.where(yy <= half_level)[0]
            if below.size:
                half_life_after_peak = float(tt[below[0]] - t_peak)

        if len(t_c) and len(mu_w):
            after = t_c >= t_peak
            neg = after & (mu_w < 0) & np.isfinite(mu_w)
            idxs = np.where(neg)[0]
            if idxs.size:
                decline_onset_time = float(t_c[idxs[0]])
            max_decline_rate = float(np.nanmin(mu_w[neg])) if np.any(neg) else nan

    stationary_duration = float(t_max - time_to_plateau) if np.isfinite(time_to_plateau) else nan

    if np.isfinite(exp_start) and np.isfinite(exp_end):
        y_start = float(np.interp(exp_start, t, y))
        y_end = float(np.interp(exp_end, t, y))
        exp_OD_gain = float(y_end - y_start)
    else:
        exp_OD_gain = nan

    if np.isfinite(duration_h) and len(t) >= 3:
        t_late0 = t_min + 0.85 * duration_h
        mask_late = t >= t_late0
        if mask_late.sum() >= 2:
            tt = t[mask_late]
            yy = y[mask_late]
            A = np.vstack([tt, np.ones_like(tt)]).T
            coef, *_ = np.linalg.lstsq(A, yy, rcond=None)
            late_OD_slope = float(coef[0])
    
    decline_fraction = float((max_OD - Endpoint_biomass) / max_OD) if max_OD > 0 else nan

    beta = 0.7
    if max_OD > 0:
        above = t[y >= beta * max_OD]
        if above.size:
            persistence_time = float(above[-1] - above[0])

    y_bc = np.maximum(y - od0, 0.0)
    OD_time_skewness = _weighted_time_skewness(t, y_bc)

    OD_entropy = _shannon_entropy(np.maximum(y, 0))
    inc_auc = (y[1:] + y[:-1]) / 2 * np.diff(t) if len(t) >= 2 else np.array([])
    AUC_entropy = _shannon_entropy(np.maximum(inc_auc, 0))
    OD_centroid_time = float(np.nansum(t * y) / np.nansum(y)) if np.nansum(y) > 0 else nan

    n_phases = second_mu_max = diauxic_lag = nan
    if len(mu_w) and np.any(np.isfinite(mu_w)):
        peaks = _local_maxima(mu_w)
        peaks = peaks[(mu_w[peaks] > 0) & np.isfinite(mu_w[peaks])]
        if peaks.size:
            heights = mu_w[peaks]
            order_peaks = np.argsort(heights)[::-1]
            peaks = peaks[order_peaks]
            n_phases = float(len(peaks))
            if len(peaks) >= 2:
                second_mu_max = float(mu_w[peaks[1]])
                t1p, t2p = float(t_c[peaks[0]]), float(t_c[peaks[1]])
                diauxic_lag = float(abs(t2p - t1p))

    initial_mu = nan
    if len(t_c) and np.isfinite(t_detect):
        mask0 = (t_c >= t_detect) & (t_c <= t_detect + 2.0) & np.isfinite(mu_w)
        if np.any(mask0):
            initial_mu = float(np.nanmean(mu_w[mask0]))

    exp_R2 = nan
    if len(mu_w) and np.any(good_mu):
        if np.isfinite(exp_start) and np.isfinite(exp_end) and exp_end > exp_start:
            mask_exp = (t >= exp_start) & (t <= exp_end)
            if mask_exp.sum() >= 3:
                yy = np.maximum(y[mask_exp], OD_EPS)
                xx = t[mask_exp]
                logy = np.log(yy)
                A = np.vstack([xx, np.ones_like(xx)]).T
                coef, *_ = np.linalg.lstsq(A, logy, rcond=None)
                pred = A @ coef
                ss_res = float(np.sum((logy - pred) ** 2))
                ss_tot = float(np.sum((logy - np.mean(logy)) ** 2))
                exp_R2 = float(1 - ss_res / ss_tot) if ss_tot > 0 else nan

    fit = _fit_gompertz(t, y)

    return {
        "max_mu": max_mu,
        "mean_mu": mean_mu,
        "max_dODdt": max_dODdt,
        "max_OD": max_OD,
        "Endpoint_biomass": Endpoint_biomass,
        "lag_phase": lag_phase,
        "t_detect": t_detect,
        "t_lag_end": t_lag_end,
        "t10": t10,
        "t50": t50,
        "t90": t90,
        "time_max_mu": time_max_mu,
        "OD_max_mu": OD_max_mu,
        "time_max_OD / t_peak": time_max_OD,
        "t_exp_start": exp_start,
        "t_exp_end": t_exp_end,
        "Exp_phase_length": Exp_phase_length,
        "t_plateau_start": time_to_plateau,
        "t_inflection": t_inflection,
        "max_acceleration": max_acceleration,
        "max_deceleration": max_deceleration,
        "decline_onset_time": decline_onset_time,
        "max_decline_rate": max_decline_rate,
        "half_life_after_peak": half_life_after_peak,
        "AUC": AUC,
        "AUC_early": AUC_early,
        "AUC_mid": AUC_mid,
        "AUC_late": AUC_late,
        "t_AUC50": t_AUC50,
        "front_loaded_fraction": front_loaded_fraction,
        "back_loaded_fraction": back_loaded_fraction,
        "growth_timing_asymmetry": growth_timing_asymmetry,
        "n_phases": n_phases,
        "second_mu_max": second_mu_max,
        "diauxic_lag": diauxic_lag,
        "model_A (asymptote)": fit.get("model_A", nan),
        "model_shape (ν, etc.)": fit.get("model_shape", nan),
        "fit_RMSE": fit.get("fit_RMSE", nan),
        "AIC / BIC": fit.get("AIC", fit.get("BIC", nan)),
        "OD_centroid_time": OD_centroid_time,
        "OD_time_skewness": OD_time_skewness,
        "persistence_time": persistence_time,
        "plateau_mean_OD": plateau_mean,
        "post_plateau_noise": post_plateau_noise,
        "stationary_duration": stationary_duration,
        "decline_fraction": decline_fraction,
        "late_OD_slope": late_OD_slope,
        "exp_OD_gain": exp_OD_gain,
        "exp_fit_R2": exp_R2,
        "initial_mu (μ₀)": initial_mu,
        "Lag_cost_index": float(lag_phase * max_mu) if np.isfinite(lag_phase) and np.isfinite(max_mu) else nan,
        "t_peak_from_start": float(t_peak - t_min) if np.isfinite(t_peak) else nan,
        "doubling_time_min": doubling_time_min,
        "OD_entropy": OD_entropy,
        "AUC_entropy": AUC_entropy,
    }


class FullFeatureAnalyzer:
    def __init__(self, data_dir):
        self.data_dir = Path(data_dir)
        
    def load_data(self):
        print("Loading cleaned data...")
        cleaned_data_dir = self.data_dir
        self.mutant_stress = pd.read_csv(cleaned_data_dir / f"01.mutant_stress_OD_cleaned.csv", index_col=0)
        self.mutant_nonstress = pd.read_csv(cleaned_data_dir / f"02.mutant_nonstress_OD_cleaned.csv", index_col=0)
        self.ctrl_stress = pd.read_csv(cleaned_data_dir / f"03.Ctrol_stress_OD_cleaned.csv", index_col=0)
        self.ctrl_nonstress = pd.read_csv(cleaned_data_dir / f"04.Ctrol_nonstress_OD_cleaned.csv", index_col=0)
        
        self.time_points = self.mutant_stress.index.values.astype(float)
        #convert time to hours for feature calculation.
        self.time_h = self.time_points / 60.0
        
        self.genes = [col.strip() for col in self.mutant_stress.columns]
        print(f"Loaded {len(self.genes)} genes, {len(self.time_points)} time points (0-{self.time_points[-1]} min).")

    def find_control_column(self, gene):
        for col in self.ctrl_stress.columns:
            if gene in col:
                return col
        return None

    def analyze_all_features(self, output_file):
        print("Calculating all features for all conditions...")
        results = []
        
        # Conditions mapping
        conditions = {
            'mut_stress': self.mutant_stress,
            'mut_nonstress': self.mutant_nonstress,
            'con_stress': self.ctrl_stress,
            'con_nonstress': self.ctrl_nonstress
        }
        
        total = len(self.genes)
        for i, gene in enumerate(self.genes):
            if i % 100 == 0:
                print(f"Processing {i}/{total}...")
                
            gene_col = gene + '  ' if gene + '  ' in self.mutant_stress.columns else gene
            if gene_col not in self.mutant_stress.columns:
                continue
                
            ctrl_col = self.find_control_column(gene)
            if ctrl_col is None:
                continue
            
            entry = {'gene': gene}
            
            # For each condition
            for prefix, df in conditions.items():
                col = gene_col if 'mut' in prefix else ctrl_col
                if col not in df.columns:
                    continue
                
                od = df[col].values
                # Calculate features
                feats = calc_curve_features(self.time_h, od)
                
                # Add to entry with prefix
                for k, v in feats.items():
                    entry[f"{prefix}_{k}"] = v
            
            results.append(entry)
            
        print(f"Finished processing {len(results)} genes.")
        df_out = pd.DataFrame(results)
        df_out.to_csv(output_file, index=False)
        print(f"Saved results to {output_file}")
