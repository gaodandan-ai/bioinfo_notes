import pandas as pd
import numpy as np
import seaborn as sns
import matplotlib.pyplot as plt
from pathlib import Path
from scipy.cluster.hierarchy import linkage, leaves_list

def plot_dotplot(top5_file, output_dir):
    df = pd.read_csv(top5_file)
    output_dir = Path(output_dir)
    output_dir.mkdir(parents=True, exist_ok=True)
    
    # 1. 构建 Binary Matrix 用于聚类
    # Rows: Genes, Cols: Features
    # value=1 if gene is top5 in feature, else 0
    genes = sorted(df['Gene'].unique())
    features = sorted(df['Feature'].unique())
    
    matrix = pd.DataFrame(0, index=genes, columns=features)
    for _, row in df.iterrows():
        matrix.loc[row['Gene'], row['Feature']] = 1
        
    # 2. 聚类排序
    # 对行 (Genes) 聚类
    # metric='jaccard' 适合二值数据，但 'euclidean' 也行
    row_linkage = linkage(matrix, method='average', metric='euclidean')
    row_order = leaves_list(row_linkage)
    sorted_genes = matrix.index[row_order].tolist()
    
    # 对列 (Features) 聚类
    col_linkage = linkage(matrix.T, method='average', metric='euclidean')
    col_order = leaves_list(col_linkage)
    sorted_features = matrix.columns[col_order].tolist()
    
    # 3. 准备绘图数据
    plot_df = df.copy()
    
    # 将 Gene 和 Feature 转为 Categorical 类型，按聚类顺序排序
    plot_df['Gene'] = pd.Categorical(plot_df['Gene'], categories=sorted_genes, ordered=True)
    plot_df['Feature'] = pd.Categorical(plot_df['Feature'], categories=sorted_features, ordered=True)
    
    # 4. 绘图
    
    plt.figure(figsize=(12, max(12, len(genes) * 0.25)))
    
    # 绘制主图 (Scatterplot)
    # hue='Rank': 颜色代表排名 (1-5)
    # size='RF_Value': 大小代表 RF 值
    
    # 使用 scatterplot
    scatter = sns.scatterplot(
        data=plot_df, 
        x='Feature', 
        y='Gene',
        hue='Rank', 
        palette='Reds_r', # Rank 1 (小) -> 深红, Rank 5 (大) -> 浅红
        size='RF_Value',
        sizes=(50, 400), 
        alpha=0.9,
        edgecolor='black',
        linewidth=0.5,
        zorder=5,
        legend='brief'
    )
    
    # 移出图例
    plt.legend(bbox_to_anchor=(1.02, 1), loc='upper left', borderaxespad=0, title='')
    
    plt.title(f'', fontsize=16, pad=20)
    plt.xlabel('', fontsize=12, fontweight='bold')
    plt.ylabel('', fontsize=12, fontweight='bold')
    
    # 调整轴标签
    plt.xticks(rotation=90, ha='right', fontsize=12)
    plt.yticks(fontsize=12)
    
    # 去掉所有边框 (Spines)，只留左边和下边？或者保留方框但去掉网格
    sns.despine(top=False, right=False, left=False, bottom=False)
    
    # 调整边距
    plt.tight_layout()
    
    output_png = output_dir / 'strain_feature_dotplot.png'
    output_pdf = output_dir / 'strain_feature_dotplot.pdf'
    
    plt.savefig(output_png, dpi=300, bbox_inches='tight')
    plt.savefig(output_pdf, bbox_inches='tight')
    print(f"Clean Dot plots saved to:\n  {output_png}\n  {output_pdf}")

if __name__ == "__main__":
    import argparse
    parser = argparse.ArgumentParser()
    parser.add_argument('--top5_file', type=str, required=True)
    parser.add_argument('--output_dir', type=str, required=True)
    args = parser.parse_args()
    
    plot_dotplot(args.top5_file, args.output_dir)
