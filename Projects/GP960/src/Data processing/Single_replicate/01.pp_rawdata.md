```python
#!/usr/bin/env python3
# -*- coding: utf-8 -*-
"""
突变菌株胁迫条件，数据处理脚本
处理突变菌株【基因过表达/TF失活/其他突变菌株】和正常对照菌株在两种条件conditon1和condition2【胁迫条件和正常条件下】的实验数据处理,生成四个CSV文件:

输入：
1. 原始实验数据文件夹，包含plate文件（Excel格式）
2. plate-gene映射文件（Excel格式）
3. 输出结果文件夹路径
4. condition1标签（例如: stress 或 nonstress）
5. condition2标签（例如: stress 或 nonstress）

输出：
1. 01.mutant_{c1_label}_OD.csv - 突变菌株condition1条件下的OD数据
2. 02.mutant_{c2_label}_OD.csv - 突变菌株condition2条件下的OD数据  
3. 03.Ctrol_{c1_label}_OD.csv - 空质粒对照condition1条件下的OD数据
4. 04.Ctrol_{c2_label}_OD.csv - 空质粒对照condition2条件下的OD数据


实验设计：
- 每个原始plate通过3个MTP实现:MTP1(1-6), MTP2(7-12), MTP3(last)
- MTP1: 原始plate A-G 1-6列 -> A-G行1-6列(condition1), A-G行7-12列(condition2)
- MTP2: 原始plate A-G 7-12列 -> A-G行1-6列(condition1), A-G行7-12列(condition2)
- MTP3: 原始plate H行 1-12列 -> A-F行1-2列对应原始H1-12(condition1), A-F行7-8列对应原始H1-12(condition2)
        H1->A1, H2->B1, H3->C1, H4->D1, H5->E1, H6->F1, H7->A2, H8->B2, H9->C2, H10->D2, H11->E2, H12->F2(condition1)
        H1->A7, H2->B7, H3->C7, H4->D7, H5->E7, H6->F7 ,H7->A8, H8->B8, H9->C8, H10->D8, H11->E8, H12->F8(condition2)

对照组设计逻辑：
A-G行1-6列 :condition1对照组是MTP1的H1-3,condition2对照组是MTP1的H7-9
A-G行7-12列 :condition1对照组是MTP2的H1-3,condition2对照组是MTP2的H7-9
H行1-12列 :condition1对照组是MTP3的H1-3,condition2对照组是MTP3的H7-9
"""

import pandas as pd
import numpy as np
import os
from pathlib import Path
import argparse

def load_plate_gene_mapping(mapping_file):
    """加载plate-gene映射关系"""
    df = pd.read_excel(mapping_file)
    # 创建位置到基因的映射字典
    position_to_gene = {}
    for _, row in df.iterrows():
        position = row['location']
        gene = row['gene']
        position_to_gene[position] = gene
    return position_to_gene

def extract_gene_data_from_mtp(df_mtp, position_to_gene, plate_num, mtp_type):
    """从单个MTP sheet中提取基因数据"""
    # 检查DataFrame是否有足够的列
    if df_mtp.shape[1] < 2:
        print(f"警告: {mtp_type} sheet 列数不足，跳过处理")
        return {}, {}, df_mtp.iloc[:, 0].dropna() if df_mtp.shape[1] > 0 else []
    
    # 获取时间列
    time_col = df_mtp.iloc[:, 1].dropna()
    
    gene_condition2_data = {}
    gene_condition1_data = {}
    
    if mtp_type in ['MTP1', 'MTP2']:
        # MTP1处理1-6列，MTP2处理7-12列
        col_offset = 0 if mtp_type == 'MTP1' else 6
        
        # 处理A-G行的基因数据
        for row_idx, row_letter in enumerate(['A', 'B', 'C', 'D', 'E', 'F', 'G']):
            for col_idx in range(6):  # 1-6列
                # 原始plate中的位置
                original_col = col_idx + 1 + col_offset  # MTP1: 1-6, MTP2: 7-12
                original_pos = f"{plate_num}{row_letter}{original_col}"
                
                if original_pos in position_to_gene:
                    gene = position_to_gene[original_pos]
                    
                    # condition1 条件 (MTP中的1-6列) - 根据正确的实验设计
                    condition1_col = f"{row_letter}{col_idx + 1}"
                    if condition1_col in df_mtp.columns:
                        condition1_data = df_mtp[condition1_col].dropna()
                        if len(condition1_data) >= len(time_col):
                            gene_condition1_data[gene] = condition1_data.iloc[:len(time_col)].values
                    
                    # condition2条件 (MTP中的7-12列) - 根据正确的实验设计
                    condition2_col = f"{row_letter}{col_idx + 7}"
                    if condition2_col in df_mtp.columns:
                        condition2_data = df_mtp[condition2_col].dropna()
                        if len(condition2_data) >= len(time_col):
                            gene_condition2_data[gene] = condition2_data.iloc[:len(time_col)].values
    
    elif mtp_type == 'MTP3':
        # MTP3处理原始plate的H行
        # 正确的转置映射关系:
        # condition1 : H1->A1, H2->B1, H3->C1, H4->D1, H5->E1, H6->F1, H7->A2, H8->B2, H9->C2, H10->D2, H11->E2, H12->F2
        # condition2: H1->A7, H2->B7, H3->C7, H4->D7, H5->E7, H6->F7, H7->A8, H8->B8, H9->C8, H10->D8, H11->E8, H12->F8
        
        for h_idx in range(12):  # H1-H12
            original_pos = f"{plate_num}H{h_idx + 1}"
            if original_pos in position_to_gene:
                gene = position_to_gene[original_pos]
                
                # 计算转置后的位置
                if h_idx < 6:  # H1-H6
                    row_letter = chr(ord('A') + h_idx)  # A-F
                    condition1_col = f"{row_letter}1"
                    condition2_col = f"{row_letter}7"
                else:  # H7-H12
                    row_letter = chr(ord('A') + (h_idx - 6))  # A-F
                    condition1_col = f"{row_letter}2"
                    condition2_col = f"{row_letter}8"
                
                # condition1 条件
                if condition1_col in df_mtp.columns:
                    condition1_data = df_mtp[condition1_col].dropna()
                    if len(condition1_data) >= len(time_col):
                        gene_condition1_data[gene] = condition1_data.iloc[:len(time_col)].values
                
                # condition2条件
                if condition2_col in df_mtp.columns:
                    condition2_data = df_mtp[condition2_col].dropna()
                    if len(condition2_data) >= len(time_col):
                        gene_condition2_data[gene] = condition2_data.iloc[:len(time_col)].values
    
    return gene_condition2_data, gene_condition1_data, time_col.values

def extract_control_data_from_mtp(df_mtp, position_to_gene, plate_num, mtp_type):
    """从MTP sheet中提取对照数据 - 优化格式，生成18列，包含具体基因名字"""
    # 检查DataFrame是否有足够的列
    if df_mtp.shape[1] < 2:
        print(f"警告: {mtp_type} sheet 列数不足，跳过对照数据处理")
        return {}, {}, []
    
    # 获取时间列
    time_col = df_mtp.iloc[:, 1].dropna()
    
    control_condition2_data = {}
    control_condition1_data = {}
    
    if mtp_type == 'MTP1':
        # MTP1的对照逻辑：为A-G行1-6列的基因提供对照
        # H1-3对应A-G行1-6列的基因（condition1 条件）
        # H7-9对应A-G行1-6列的基因（condition2条件）
        
        # 收集A-G行1-6列的基因
        target_genes = []
        for row_idx in range(7):  # A-G行
            row_letter = chr(ord('A') + row_idx)
            for col_idx in range(6):  # 1-6列
                original_pos = f"{plate_num}{row_letter}{col_idx + 1}"
                if original_pos in position_to_gene:
                    target_genes.append(position_to_gene[original_pos])
        
        # 生成基因列表字符串（用分号分隔）
        gene_list = ';'.join(target_genes) if target_genes else "NoGenes"
        
        # 提取H1-3的对照数据，对应condition1条件
        for i, col in enumerate(['H1', 'H2', 'H3'], 1):
            if col in df_mtp.columns:
                data = df_mtp[col].dropna()
                if len(data) >= len(time_col):
                    control_data = data.iloc[:len(time_col)].values
                    # 使用新的列名格式，包含具体基因名字
                    col_name = f"ctrol{i}-{gene_list}"
                    control_condition1_data[col_name] = control_data
        
        # 提取H7-9的对照数据，对应condition2条件
        for i, col in enumerate(['H7', 'H8', 'H9'], 1):
            if col in df_mtp.columns:
                data = df_mtp[col].dropna()
                if len(data) >= len(time_col):
                    control_data = data.iloc[:len(time_col)].values
                    # 使用新的列名格式，包含具体基因名字
                    col_name = f"ctrol{i}-{gene_list}"
                    control_condition2_data[col_name] = control_data
    
    elif mtp_type == 'MTP2':
        # MTP2的对照逻辑：为A-G行7-12列的基因提供对照
        # H1-3对应A-G行7-12列的基因（condition1条件）
        # H7-9对应A-G行7-12列的基因（condition2条件）
        
        # 收集A-G行7-12列的基因
        target_genes = []
        for row_idx in range(7):  # A-G行
            row_letter = chr(ord('A') + row_idx)
            for col_idx in range(6):  # 7-12列
                original_pos = f"{plate_num}{row_letter}{col_idx + 7}"
                if original_pos in position_to_gene:
                    target_genes.append(position_to_gene[original_pos])
        
        # 生成基因列表字符串（用分号分隔）
        gene_list = ';'.join(target_genes) if target_genes else "NoGenes"
        
        # 提取H1-3的对照数据，对应condition1条件
        for i, col in enumerate(['H1', 'H2', 'H3'], 1):
            if col in df_mtp.columns:
                data = df_mtp[col].dropna()
                if len(data) >= len(time_col):
                    control_data = data.iloc[:len(time_col)].values
                    # 使用新的列名格式，包含具体基因名字
                    col_name = f"ctrol{i}-{gene_list}"
                    control_condition1_data[col_name] = control_data
        
        # 提取H7-9的对照数据，对应condition2条件
        for i, col in enumerate(['H7', 'H8', 'H9'], 1):
            if col in df_mtp.columns:
                data = df_mtp[col].dropna()
                if len(data) >= len(time_col):
                    control_data = data.iloc[:len(time_col)].values
                    # 使用新的列名格式，包含具体基因名字
                    col_name = f"ctrol{i}-{gene_list}"
                    control_condition2_data[col_name] = control_data
    
    elif mtp_type == 'MTP3':
        # MTP3的对照逻辑：为H行1-12列的基因提供对照
        # H1-3对应H行1-12列的基因（condition1条件）
        # H7-9对应H行1-12列的基因（condition2条件）
        
        # 收集H行1-12列的基因
        target_genes = []
        for col_idx in range(12):  # 1-12列
            original_pos = f"{plate_num}H{col_idx + 1}"
            if original_pos in position_to_gene:
                target_genes.append(position_to_gene[original_pos])
        
        # 生成基因列表字符串（用分号分隔）
        gene_list = ';'.join(target_genes) if target_genes else "NoGenes"
        
        # 提取H1-3的对照数据，对应condition1条件
        for i, col in enumerate(['H1', 'H2', 'H3'], 1):
            if col in df_mtp.columns:
                data = df_mtp[col].dropna()
                if len(data) >= len(time_col):
                    control_data = data.iloc[:len(time_col)].values
                    # 使用新的列名格式，包含具体基因名字
                    col_name = f"ctrol{i}-{gene_list}"
                    control_condition1_data[col_name] = control_data
        
        # 提取H7-9的对照数据，对应condition2条件
        for i, col in enumerate(['H7', 'H8', 'H9'], 1):
            if col in df_mtp.columns:
                data = df_mtp[col].dropna()
                if len(data) >= len(time_col):
                    control_data = data.iloc[:len(time_col)].values
                    # 使用新的列名格式，包含具体基因名字
                    col_name = f"ctrol{i}-{gene_list}"
                    control_condition2_data[col_name] = control_data
    
    return control_condition2_data, control_condition1_data, time_col.values

def process_single_plate(plate_file, position_to_gene, plate_num):
    """处理单个plate文件"""
    print(f"处理 {plate_file}...")
    
    try:
        # 读取三个MTP sheet
        df_mtp1 = pd.read_excel(plate_file, sheet_name='MTP1(1-6)')
        df_mtp2 = pd.read_excel(plate_file, sheet_name='MTP2(7-12)')  
        df_mtp3 = pd.read_excel(plate_file, sheet_name='MTP3(last)')
    except Exception as e:
        print(f"错误: 无法读取文件 {plate_file}: {e}")
        return {}, {}, {}, {}, None
    
    all_gene_condition2 = {}
    all_gene_condition1 = {}
    all_control_condition2 = {}
    all_control_condition1 = {}
    time_points = None
    
    # 处理MTP1
    gene_condition2, gene_condition1, time_col = extract_gene_data_from_mtp(
        df_mtp1, position_to_gene, plate_num, 'MTP1')
    all_gene_condition2.update(gene_condition2)
    all_gene_condition1.update(gene_condition1)
    time_points = time_col
    
    control_condition2, control_condition1, _ = extract_control_data_from_mtp(
        df_mtp1, position_to_gene, plate_num, 'MTP1')
    all_control_condition2.update(control_condition2)
    all_control_condition1.update(control_condition1)
    
    # 处理MTP2
    gene_condition2, gene_condition1, _ = extract_gene_data_from_mtp(
        df_mtp2, position_to_gene, plate_num, 'MTP2')
    all_gene_condition2.update(gene_condition2)
    all_gene_condition1.update(gene_condition1)
    
    control_condition2, control_condition1, _ = extract_control_data_from_mtp(
        df_mtp2, position_to_gene, plate_num, 'MTP2')
    all_control_condition2.update(control_condition2)
    all_control_condition1.update(control_condition1)
    
    # 处理MTP3
    gene_condition2, gene_condition1, _ = extract_gene_data_from_mtp(
        df_mtp3, position_to_gene, plate_num, 'MTP3')
    all_gene_condition2.update(gene_condition2)
    all_gene_condition1.update(gene_condition1)
    
    control_condition2, control_condition1, _ = extract_control_data_from_mtp(
        df_mtp3, position_to_gene, plate_num, 'MTP3')
    all_control_condition2.update(control_condition2)
    all_control_condition1.update(control_condition1)
    
    return all_gene_condition2, all_gene_condition1, all_control_condition2, all_control_condition1, time_points

def main():
    import argparse
    parser = argparse.ArgumentParser(description='处理高通量生长曲线数据')
    parser.add_argument('--rawdata_dir', type=str, required=True,
                        help='原始实验数据目录，如: /data/zuoll/1.project/02.GOE/GP960-growth-screen-pipeline/data/raw/single_replicate/highmethanol')
    parser.add_argument('--plate_file', type=str, required=True,
                        help='实验对应关系文件，如: /data/zuoll/1.project/02.GOE/GP960-growth-screen-pipeline/data/raw/single_replicate/plate-gene-mapping.xlsx')
    parser.add_argument('--result_dir', type=str, required=True,
                        help='处理后的结果目录，如: /data/zuoll/1.project/02.GOE/GP960-growth-screen-pipeline/data/results/single_replicate/highmethanol')
    parser.add_argument('--c1_label', type=str, default='stress',
                    choices=['stress', 'nonstress'],  # 限制只能选择这两个值
                    help='Condition 1 的标签: stress (胁迫) 或 nonstress (无胁迫)')

    parser.add_argument('--c2_label', type=str, default='nonstress',
                    choices=['stress', 'nonstress'],  # 同样限制选择范围
                    help='Condition 2 的标签: stress (胁迫) 或 nonstress (无胁迫)')
    args = parser.parse_args()
    
    # 设置路径
    rawdata_dir = Path(args.rawdata_dir)
    plate_file_path = Path(args.plate_file)
    result_dir = Path(args.result_dir)
    
    # 创建结果目录
    result_dir.mkdir(parents=True, exist_ok=True)
    
    # 加载plate-gene映射
    if not plate_file_path.exists():
        print(f"错误: 找不到 mapping 文件 {plate_file_path}")
        return

    position_to_gene = load_plate_gene_mapping(plate_file_path)
    print(f"加载了 {len(position_to_gene)} 个基因映射关系")
    
    # 初始化数据收集器
    all_gene_condition2_data = {}
    all_gene_condition1_data = {}
    all_control_condition2_data = {}
    all_control_condition1_data = {}
    time_points = None
    
    # 自动识别plate文件
    # 查找符合模式 {plate_num:02d}-{args.condition}.xlsx 的文件
    # 或者如果不指定 condition 参数，查找所有 .xlsx 文件并尝试解析
    
    # 移除 condition 参数依赖，改用自动搜索
    xlsx_files = sorted(list(rawdata_dir.glob('*.xlsx')))
    
    # 过滤掉 mapping 文件
    xlsx_files = [f for f in xlsx_files if "mapping" not in f.name.lower()]
    
    if not xlsx_files:
        print(f"警告: 在 {rawdata_dir} 未找到数据文件")
        return

    print(f"找到 {len(xlsx_files)} 个数据文件: {[f.name for f in xlsx_files]}")
    
    # 处理找到的每个文件
    for plate_file in xlsx_files:
        # 尝试从文件名提取 plate_num (假设文件名以数字开头，如 01-xxx.xlsx)
        try:
            plate_prefix = plate_file.name.split('-')[0]
            plate_num = int(plate_prefix)
        except ValueError:
            print(f"警告: 无法从文件名 {plate_file.name} 提取 plate 编号，跳过处理")
            continue
            
        gene_condition2, gene_condition1, control_condition2, control_condition1, times = process_single_plate(
            plate_file, position_to_gene, plate_num)
        
        # 合并数据
        all_gene_condition2_data.update(gene_condition2)
        all_gene_condition1_data.update(gene_condition1)
        all_control_condition2_data.update(control_condition2)
        all_control_condition1_data.update(control_condition1)
        
        if time_points is None:
            time_points = times
            
        print(f"从 {plate_file.name} 提取了 {len(gene_condition2)} 个condition2基因, {len(gene_condition1)} 个condition1 基因")
    
    # 创建DataFrame并保存
    print("生成CSV文件...")
    print(f"总共收集到: condition1基因 {len(all_gene_condition1_data)}, condition2 基因 {len(all_gene_condition2_data)}")
    print(f"对照数据: condition1 {len(all_control_condition1_data)}, condition2  {len(all_control_condition2_data)}")
    
    # 1. mutant_condition1_OD.csv (使用c1_label)
    if all_gene_condition1_data:
        df_gene_condition1 = pd.DataFrame(all_gene_condition1_data, index=time_points)
        filename = f"01.mutant_{args.c1_label}_OD.csv"
        df_gene_condition1.to_csv(result_dir / filename)
        print(f"生成 {filename}: {df_gene_condition1.shape}")
    
    # 2. mutant_condition2_OD.csv (使用c2_label)
    if all_gene_condition2_data:
        df_gene_condition2 = pd.DataFrame(all_gene_condition2_data, index=time_points)
        filename = f"02.mutant_{args.c2_label}_OD.csv"
        df_gene_condition2.to_csv(result_dir / filename) 
        print(f"生成 {filename}: {df_gene_condition2.shape}")
    
    # 3. Ctrol_condition1_OD.csv (使用c1_label)
    if all_control_condition1_data:
        df_control_condition1 = pd.DataFrame(all_control_condition1_data, index=time_points)
        filename = f"03.Ctrol_{args.c1_label}_OD.csv"
        df_control_condition1.to_csv(result_dir / filename)  
        print(f"生成 {filename}: {df_control_condition1.shape}")

    # 4. Ctrol_condition2_OD.csv (使用c2_label)
    if all_control_condition2_data:
        df_control_condition2 = pd.DataFrame(all_control_condition2_data, index=time_points)
        filename = f"04.Ctrol_{args.c2_label}_OD.csv"
        df_control_condition2.to_csv(result_dir / filename)
        print(f"生成 {filename}: {df_control_condition2.shape}")

if __name__ == "__main__":
    main()
```
