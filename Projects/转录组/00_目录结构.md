# 重新生成含通路名称的谷氨酸棒杆菌gmt文件（彻底解决NA）
if (!require("KEGGREST", quietly = TRUE)) BiocManager::install("KEGGREST")
library(KEGGREST)
library(dplyr)

# 谷氨酸棒杆菌ATCC13032的KEGG物种代码：cgl（固定）
kegg_species <- "cgl"
# 新gmt文件保存路径（与你原有路径一致，直接覆盖）
new_gmt_file <- "G:\\thermal_wqz\\Cgl_ATCC13032_KEGG_pathways.gmt"

# 1. 获取该菌株所有KEGG通路ID和名称
cat("🔍 从KEGG获取谷氨酸棒杆菌通路信息...\n")
kegg_pathways <- keggList("pathway", kegg_species)
# 清理通路ID和名称（去除前缀，适配gmt）
pathway_clean <- data.frame(
  path_id = sub("^path:", "", names(kegg_pathways)),  # 通路ID：cgl00010
  path_name = sub(paste0("^", kegg_species, " "), "", kegg_pathways)  # 通路名称：Glycolysis/Gluconeogenesis
)
cat(paste0("✅ 获取到 ", nrow(pathway_clean), " 个KEGG通路\n"))

# 2. 批量获取每个通路的基因（locus tag：CglXXXX，与你的数据ID匹配）
pathway_gene_list <- list()
for (i in 1:nrow(pathway_clean)) {
  path_id <- pathway_clean$path_id[i]
  tryCatch({
    # 从KEGG获取通路基因
    path_info <- keggGet(paste0("path:", path_id))[[1]]$GENE
    if (is.null(path_info)) next
    # 提取locus tag（KEGG返回格式：Cgl0001; gapA; ...，取第一部分）
    locus_tags <- sapply(strsplit(path_info, "; "), function(x) x[1])
    pathway_gene_list[[path_id]] <- unique(locus_tags)
  }, error = function(e) next)
}
cat(paste0("✅ 成功解析 ", length(pathway_gene_list), " 个通路的基因集\n"))

# 3. 生成标准gmt文件（第1列ID，第2列名称，第3列起基因）
con <- file(new_gmt_file, "w")
for (path_id in names(pathway_gene_list)) {
  path_name <- pathway_clean$path_name[pathway_clean$path_id == path_id]
  genes <- pathway_gene_list[[path_id]]
  # 按gmt格式写入：ID\t名称\t基因1\t基因2...
  writeLines(paste(c(path_id, path_name, genes), collapse = "\t"), con)
}
close(con)

cat(paste0("🎉 标准gmt文件生成完成！路径：", new_gmt_file, "\n"))
cat("📌 包含：通路ID+完整KEGG名称+基因ID，无NA值\n")
